16  package net.oneandone.sushi.util;

18  import java.util.ArrayList;
19  import java.util.List;
20  
21  public class Lcs<T> {
22      public static <T> List<T> compute(List<T> vert, List<T> hor) {
23          List<T>[] previous;
24          List<T>[] current;
25          
26          previous = new List[hor.size() + 1];
27          for (int i = 0; i < previous.length; i++) {
28              previous[i] = new ArrayList<T>();
29          }
30  
31          for (T value : vert) {
32              current = new List[hor.size() + 1];
33              current[0] = new ArrayList<T>();
34              for (int i = 1; i < current.length; i++) {
35                  if (value.equals(hor.get(i - 1))) {
36                      current[i] = append(previous[i - 1], value);
37                  } else {
38                      current[i] = longest(current[i - 1], previous[i]);
39                  }
40              }
41              previous = current;
42          }
43          return previous[previous.length - 1];
44      }
45      
46      private static <T> List<T> append(List<T> lst, T value) {
47          List<T> result;
48          
49          result = new ArrayList<T>(lst.size() + 1);
50          result.addAll(lst);
51          result.add(value);
52          return result;
53      }
54      
55      private static <T> List<T> longest(List<T> a, List<T> b) {
56          return new ArrayList<T>(a.size() >= b.size() ? a : b);
57      }
