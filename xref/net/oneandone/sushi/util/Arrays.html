16  package net.oneandone.sushi.util;

18  import java.lang.reflect.Array;
19  
20  /**
21   * Additional functionality over java.util.Arrays.
22   */
23  
24  public class Arrays {
25      public static Object[] append(Class<?> cl, Object[] left, Object[] right) {
26          Object[] result;
27  
28          result = (Object[]) Array.newInstance(cl, left.length + right.length);
29          System.arraycopy(left, 0, result, 0, left.length);
30          System.arraycopy(right, 0, result, left.length, right.length);
31          return result;
32      }
33  
34      public static Object[] cons(Class<?> cl, Object left, Object[] right) {
35          Object[] result;
36  
37          result = (Object[]) Array.newInstance(cl, 1 + right.length);
38          result[0] = left;
39          System.arraycopy(right, 0, result, 1, right.length);
40          return result;
41      }
42  
43      public static Object[] pair(Class<?> cl, Object left, Object right) {
44          Object[] result;
45  
46          result = (Object[]) Array.newInstance(cl, 2);
47          result[0] = left;
48          result[1] = right;
49          return result;
50      }
51  
52      /**
53       * Return an array class object with component as component type.
54       * The API description for java.lang.Class (Java 2) mentions
55       * that forName can be called with an Array name, but neither
56       * Class.forName("java.lang.String[]") nor
57       * Class.forName("[java.lang.String") returns an array class!?
58       */
59      public static Class<?> getArrayClass(Class<?> component) {
60          return Array.newInstance(component, 0).getClass();
61      }
62  }
