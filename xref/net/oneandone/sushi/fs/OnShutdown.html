16  package net.oneandone.sushi.fs;
17  
18  import net.oneandone.sushi.fs.file.FileNode;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.text.SimpleDateFormat;
23  import java.util.ArrayList;
24  import java.util.Date;
25  import java.util.List;
26  
27  /** 
28   * Shutdown hook to delete temporary FileNodes (in particular: directories, because 
29   * deleteAtExist is restricted to files) and execute arbirary other tasks.
30   * 
31   * The implementation is intentionally tied to FileNode, it doesn't support Nodes in general because:
32   * 1) I create temp file on disk only - I can't see a use case for other node implementations.   
33   * 2) node.delete() is might fail because server connections might already be closed
34   * 3) only java.world.File can create a temp file atomically
35   */
36  public class OnShutdown extends Thread {
37      private static OnShutdown singleton;
38  
39      /** a static singleton, because I don't want a shutdown hook for every world instance */
40      public static synchronized OnShutdown get() {
41          if (singleton == null) {
42              singleton = new OnShutdown();
43              Runtime.getRuntime().addShutdownHook(singleton);
44          }
45          return singleton;
46      }
47  
48      /** null if the exit task has already been started */
49      private List<FileNode> delete;
50  
51      private final String prefix;
52  
53      private final String suffix;
54      
55      // to generate directory names
56      private int dirNo;
57  
58      private final List<Runnable> onShutdown;
59      
60      public OnShutdown() {
61          this.delete = new ArrayList<FileNode>();
62          this.prefix = "sushi" + new SimpleDateFormat("MMdd-HHmm").format(new Date()) + "-"
63              + (System.currentTimeMillis() % 100)  + "-";
64          this.suffix = ".tmp";
65          this.dirNo = 0;
66          this.onShutdown = new ArrayList<Runnable>();
67      }
68  
69      public synchronized void onShutdown(Runnable runnable) {
70          onShutdown.add(runnable);
71      }
72      
73      //--
74  
75      public FileNode createFile(FileNode dir) throws IOException {
76          FileNode file;
77          
78          file = new FileNode(dir.getRoot(), File.createTempFile(prefix, suffix, dir.toPath().toFile()).toPath());
79          deleteAtExit(file);
80          return file;
81      }
82  
83      public FileNode createDirectory(FileNode dir) throws IOException {
84          FileNode file;
85      
86          dir.checkDirectory();
87          for (; true; dirNo++) {
88              file = dir.join(prefix + dirNo + suffix);
89              if (!file.exists()) {
90                  file.mkdir(); // do not catch IOExceptios here -- it might be "disk full" ...
91                  deleteAtExit(file);
92                  return file;
93              }
94          }
95      }
96  
97      //--
98      
99      @Override
100     public synchronized void run() {
101         List<FileNode> tmp;
102         
103         tmp = delete;
104         delete = null;
105         for (FileNode node : tmp) {
106             tryDelete(node);
107         }
108         for (Runnable runnable : onShutdown) {
109             runnable.run();
110         }
111     }
112 
113     /**
114      * Deletes all temp files created until now, without waiting for shutdown - usefull for long-running processes.
115      * Caution: the caller has to be sure that none of the files is still in use. It's usually better to explicitly
116      * delete temporary files in your application if you're know they are no longer used. */
117     public synchronized void deleteNow() {
118         for (FileNode node : delete) {
119             tryDelete(node);
120         }
121         delete.clear();
122     }
123 
124     //--
125     
126     /**
127      * @param node  file or directory
128      */
129     private synchronized void deleteAtExit(FileNode node) {
130         if (delete == null) {
131             // already exiting
132             tryDelete(node);
133         } else {
134             delete.add(node);
135         }
136     }
137 
138     private boolean tryDelete(FileNode node) {
139         try {
140             if (node.exists()) {
141                 node.deleteTree();
142             }
143             return true;
144         } catch (IOException e) {
145             e.printStackTrace();
146             return false;
147         }
148     }
