60      @Override
61      public SshNode node(URI uri, Object extra) throws NodeInstantiationException {
62          if (extra != null) {
63              throw new NodeInstantiationException(uri, "unexpected extra argument: " + extra);
64          }
65          checkHierarchical(uri);
66          try {
67              return root(uri.getAuthority()).node(getCheckedPath(uri), null);
68          } catch (JSchException | IOException e) {
69              throw new NodeInstantiationException(uri, "cannot create root", e);
70          }
71      }

81      public SshRoot root(String root, int timeout) throws JSchException, IOException {
82          int idx;
83          String host;
84          String user;
85  
86          host = root;
87          idx = host.indexOf('@');
88          if (idx == -1) {
89              user = null;
90          } else {
91              user = host.substring(0, idx);
92              host = host.substring(idx + 1);
93          }
94          return root(host, user, timeout);
95      }

97      public SshRoot root(String host, String user) throws JSchException, IOException {
98          return root(host, user, defaultTimeout);
99      }

101     /** @param user null to use current user */
102     public SshRoot root(String host, String user, int timeout) throws JSchException, IOException {
103         if (user == null) {
104             user = getWorld().getHome().getName();
105         }
106         if (jsch.getIdentityNames().isEmpty()) {
107             addDefaultIdentity();
108         }
109         return new SshRoot(this, host, user, timeout);
110     }

118     /** @param passphrase null to try to load passphrase from ~/.ssh/passphrase file */
119     public void addDefaultIdentity(String passphrase) throws IOException, JSchException {
120         Node dir;
121         Node file;
122         Node key;
123 
124         dir = getWorld().getHome().join(".ssh");
125         file = dir.join("passphrase");
126         if (passphrase == null && file.exists()) {
127             passphrase = file.readString().trim();
128         }
129         key = dir.join("id_dsa");
130         if (!key.exists()) {
131             key = dir.join("id_rsa");
132             if (!key.exists()) {
133                 key = dir.join("identity");
134             }
135         }
136         if (!key.isFile()) {
137             throw new IOException("private key not found: " + key);
138         }
139         addIdentity(key, passphrase);
140     }

142     public void addIdentity(Node privateKey, String passphrase) throws IOException, JSchException {
143         Identity identity;
144         Throwable te;
145         Class<?> clz;
146         Method m;
147         byte[] bytes;
148 
149         bytes = privateKey.readBytes();
150         // CAUTION: I cannot use
151         //   jsch.addIdentity("foo", null, null, null);
152         // because in jsch 1.48, there's no way to obtain the resulting identity and the identity.setPassphrase
153         // result.
154         try {
155             clz = Class.forName("com.jcraft.jsch.IdentityFile");
156             m = clz.getDeclaredMethod("newInstance", String.class, byte[].class, byte[].class, JSch.class);
157             m.setAccessible(true);
158             identity = (Identity) m.invoke(null, privateKey.toString(), Arrays.copyOf(bytes, bytes.length), null, jsch);
159         } catch (InvocationTargetException e) {
160             te = e.getTargetException();
161             if (te instanceof JSchException) {
162                 throw (JSchException) te;
163             } else {
164                 throw new IllegalStateException(e);
165             }
166         } catch (Exception e) {
167             throw new RuntimeException("TODO", e);
168         }
169         if (passphrase != null) {
170             if (!identity.isEncrypted()) {
171                 throw new JSchException("unexpected passphrase");
172             }
173             if (!identity.setPassphrase(passphrase.getBytes())) {
174                 throw new JSchException("invalid passphrase");
175             }
176         } else {
177             if (!identity.setPassphrase(null)) {
178                 throw new JSchException("missing passphrase");
179             }
180         }
181         jsch.removeIdentity(identity);
182         jsch.addIdentity(identity, null);
183     }
