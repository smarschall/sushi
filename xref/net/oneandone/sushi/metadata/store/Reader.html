16  package net.oneandone.sushi.metadata.store;
17  
18  import net.oneandone.sushi.metadata.Cardinality;
19  import net.oneandone.sushi.metadata.ComplexType;
20  import net.oneandone.sushi.metadata.Item;
21  import net.oneandone.sushi.metadata.SimpleType;
22  import net.oneandone.sushi.metadata.SimpleTypeException;
23  import net.oneandone.sushi.metadata.Type;
24  
25  import java.util.ArrayList;
26  import java.util.Collection;
27  import java.util.Collections;
28  import java.util.List;
29  
30  public class Reader {
31      private final Store src;
32      
33      public Reader(Store src) {
34          this.src = src;
35      }
36  
37      public Object read(Type type) throws StoreException {
38          return read("", type);
39      }
40  
41      public Object read(String path, Type type) throws StoreException {
42          return read(new ArrayList<Item<?>>(), path, type);
43      }
44      
45      private Object read(List<Item<?>> parents, String path, Type type) throws StoreException {
46          ComplexType parent;
47          String data;
48          Object obj;
49          String childPath;
50          Cardinality card;
51          Collection<?> col;
52          
53          data = readValue(parents, path);
54          if (data == null) {
55              throw new StoreException(path + ": value not found");
56          }
57          if (type instanceof SimpleType) {
58              try {
59                  return ((SimpleType) type).stringToValue(data);
60              } catch (SimpleTypeException e) {
61                  throw new StoreException(path + ": invalid value: " + e.getMessage(), e);
62              }
63          } else {
64              try {
65                  type = type.getSchema().type(Class.forName(data));
66              } catch (ClassNotFoundException e) {
67                  throw new StoreException(path + ": class not found: " + data, e);
68              }
69              parent = (ComplexType) type;
70              obj = parent.newInstance();
71              for (Item item : parent.items()) {
72                  parents.add(item);
73                  childPath = join(path, item.getName());
74                  card = item.getCardinality();
75                  if (item.getCardinality() == Cardinality.SEQUENCE) {
76                      col = readIndexed(parents, childPath, item.getType());
77                  } else {
78                      col = readNormal(parents, childPath, item.getType());
79                  }
80                  if (col.size() < card.min) {
81                      throw new StoreException(childPath + ": missing values: expected " + card.min + ", got " + col.size());
82                  }
83                  if (col.size() > card.max) {
84                      throw new StoreException(childPath + ": to many values: expected " + card.max + ", got " + col.size());
85                  }
86                  item.set(obj, col);
87                  parents.remove(parents.size() - 1);
88              }
89              return obj;
90          }
91      }
92  
93      private Collection<?> readIndexed(List<Item<?>> parents, String path, Type type) throws StoreException {
94          List<Object> col;
95          String childPath;
96          
97          col = new ArrayList<Object>();
98          for (int i = 0; true; i++) {
99              childPath = path + "[" + Integer.toString(i) + "]";
100             if (readValue(parents, childPath) == null) {
101                 return col;
102             }
103             col.add(read(parents, childPath, type));
104         }
105     }
106 
107     private Collection<?> readNormal(List<Item<?>> parents, String path, Type type) throws StoreException {
108         if (readValue(parents, path) != null) {
109             return Collections.singleton(read(parents, path, type));
110         } else {
111             return Collections.EMPTY_LIST;
112         }
113     }
114     
115     private String readValue(List<Item<?>> parents, String path) throws StoreException {
116         try {
117             return src.read(parents, path);
118         } catch (RuntimeException e) {
119             throw e;
120         } catch (Exception e) {
121             throw new StoreException(path + ": read failed: " + e.getMessage(), e);
122         }
123     }
124 
125     private static String join(String first, String second) {
126         return first.length() == 0 ? second : first + "/" + second;
127     }
