24  public class FilesystemTest {
25      private final Filesystem fs = new Filesystem(new World(), null, "foo") {
26          @Override
27          public Node node(URI uri, Object extra) throws NodeInstantiationException {
28              return null;
29          }
30      };

32      @Test
33      public void normalizeUnchanged() {
34          check("");
35          check("/");
36          check("/foo");
37          check("/bar/");
38          check("abc");
39          check("x/y/z");
40          check("nix.");
41          check("/.com");
42          check("foo./");
43          check("foo.bar");
44          check("some/path/foo.bar");
45      }
46  
47      @Test
48      public void normalizeDoubleSlash() {
49          check("/", "//");
50          check("/", "///");
51          check("/", "////");
52          check("A/B", "A//B");
53          check("A/B", "A///B");
54          check("1/2/3/", "1///2///3////");
55      }
56  
57      @Test
58      public void normalizeDots() {
59          check("", ".");
60          check("a", "a/.");
61          check("A", "./A");
62          check("a/b", "a/./b");
63      }
64  
65      @Test
66      public void normalizeDoubleDots() {
67          check("", "a/..");
68          check("", "a/../");
69          check("b", "a/../b");
70          check("b/", "a/../b/");
71          check("/b", "/a/../b");
72          check("A/B", "a/b/../../A/B");
73          check("a/x/Z", "a/b/../x/y/z/../../Z");
74      }
75  
76      private void check(String unchanged) {
77          check(unchanged, unchanged);
78      }
79  
80      private void check(String expected, String path) {
81          StringBuilder builder;
82  
83          builder = new StringBuilder(path);
84          fs.normalize(builder);
85          assertEquals(expected, builder.toString());
86      }
