24  public class SelectorTest {
25      private static final Builder BUILDER = new Builder();
26      private static final Selector SELECTOR = new Selector();

28      @Test
29      public void getIntNormal() throws XmlException {
30          assertEquals(42, getInt("42"));
31      }
32  
33      @Test
34      public void getIntNagative() throws XmlException {
35          assertEquals(-1, getInt("-1"));
36      }
37      @Test
38      public void getIntMax() throws XmlException {
39          assertEquals(Integer.MAX_VALUE, getInt("" + Integer.MAX_VALUE));
40      }
41      
42      @Test(expected=XmlException.class)
43      public void getOverflow() throws XmlException {
44          getInt("" + (((long) Integer.MAX_VALUE) + 1));
45      }
46  
47      @Test
48      public void getLong() throws XmlException {
49          Document doc;
50          long v;
51          
52          v = Integer.MAX_VALUE;
53          v++;
54          assertTrue(v > Integer.MAX_VALUE);
55          doc = BUILDER.literal("<root>" + v + "</root>");        
56          assertEquals(v, SELECTOR.longer(doc.getDocumentElement(), "."));
57      }
58  
59      private int getInt(String str) throws XmlException {
60          Document doc;
61          
62          doc = BUILDER.literal("<root>" + str + "</root>");        
63          return SELECTOR.integer(doc.getDocumentElement(), ".");
64      }

68      @Test 
69      public void string() throws XmlException {
70          Document doc;
71          
72          doc = BUILDER.literal(
73                  "<published time='1173806109'>" +
74                  "  <date format='RFC2822'>Tue, 13 Mar 2007 18:15:09 +0100</date>" +
75                  "  <date format='RFC3339'>2007-03-13T18:15:09+01:00</date>" +
76                  "</published>");
77          assertEquals("Tue, 13 Mar 2007 18:15:09 +0100", SELECTOR.string(doc, "published/date[@format='RFC2822']"));
78      }
